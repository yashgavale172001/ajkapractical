const http = require('http')
const fs = require('fs')
const fileContent = fs.readFileSync('index.html')

const server = http.createServer((req, res)=> {
    res.writeHead(200, {'Content-type' : 'text/html'});
    res.end(fileContent)
})

server.listen(8080, ()=>{
    console.log('Listening on the port 8080')
})
